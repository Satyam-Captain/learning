import ssl
from ldap3 import Server, Connection, ALL, Tls

def authenticate_user(username, password):
    """
    Validates credentials against Airbus LDAP proxy.
    Verified format: Just the username (e.g., stiwxamh).
    """
    LDAP_SERVER = 'ldapproxy-vip.eu.airbus.corp' [cite: 1]
    LDAP_PORT = 636 [cite: 1]
    
    try:
        # Secure TLS configuration is mandatory for port 636 [cite: 2]
        tls_config = Tls(validate=ssl.CERT_NONE, version=ssl.PROTOCOL_TLSv1_2)
        server = Server(LDAP_SERVER, port=LDAP_PORT, use_ssl=True, tls=tls_config, get_info=ALL)
        
        # Connection attempt using the SIMPLE bind method [cite: 3]
        conn = Connection(server, user=username, password=password, authentication='SIMPLE')
        
        if conn.bind():
            conn.unbind()
            return True
        return False
    except Exception as e:
        print("LDAP Auth Error: {}".format(e))
        return False

from flask import Flask, flash, redirect, render_template, request, session, url_for
from core.auth_service import authenticate_user 

app = Flask(__name__)
# Uses environment variable or a default dev key for sessions 
app.secret_key = os.environ.get("FLASK_SECRET_KEY", "opscore-dev-insecure-change-for-production")

@app.before_request
def require_login():
    """Protect all routes except 'login' and 'static' files."""
    allowed_routes = ['login', 'static']
    if request.endpoint not in allowed_routes and 'user' not in session:
        return redirect(url_for('login'))

@app.route('/login', methods=['GET', 'POST'])
def login():
    if request.method == 'POST':
        username = request.form.get('username')
        password = request.form.get('password')
        
        if authenticate_user(username, password):
            session['user'] = username
            flash("Logged in as {}".format(username), "ok")
            return redirect(url_for('index'))
        else:
            flash("Invalid Airbus ID or Password", "error")
            
    return render_template('login.html')

@app.route('/logout')
def logout():
    session.pop('user', None)
    return redirect(url_for('login'))
